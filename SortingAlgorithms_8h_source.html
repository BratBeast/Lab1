20template <typename T>
21class InsertionSort : public SortingAlgorithm<T> {
22public:
23 void sort(List<T>& list, std::function<bool(const T& a, const T& b)> comparator) override {
24 for (size_t i = 1; i < list.size(); i++) {
25 T key = list.get(i);
26 int j = i - 1;
27 while (j >= 0 && comparator(key, list.get(j))) {
28 list.set(j + 1, list.get(j));
29 j--;
30 }
31 list.set(j + 1, key);
32 }
33 }

38public:
39 void sort(List<T>& list, std::function<bool(const T&, const T&)> comparator) override {
40 for (size_t i = 0; i < list.size() - 1; i++) {
41 size_t min_idx = i;
42 for (size_t j = i + 1; j < list.size(); j++) {
43 if (comparator(list.get(j), list.get(min_idx))) {
44 min_idx = j;

54template <typename T>
55class QuickSort : public SortingAlgorithm<T> {
56public:
57 void sort(List<T>& list, std::function<bool(const T&, const T&)> comp) override {
58 if (list.size() > 1) quickSortRecursive(list, 0, list.size() - 1, comp);
59 }
60private:
61 void quickSortRecursive(List<T>& list, int low, int high, std::function<bool(const T&, const T&)> comp) {
62 if (low < high) {
63 int pi = partition(list, low, high, comp);
64 quickSortRecursive(list, low, pi - 1, comp);
65 quickSortRecursive(list, pi + 1, high, comp);
66 }
67 }
68 int partition(List<T>& list, int low, int high, std::function<bool(const T&, const T&)> comp) {
69 T pivot = list.get(high);
70 int i = (low - 1);
71 for (int j = low; j < high; j++) {
72 if (comp(list.get(j), pivot)) list.swap(++i, j);
73 }
74 list.swap(i + 1, high);
75 return (i + 1);
76 }
77};

79template <typename T>
80class MergeSort : public SortingAlgorithm<T> {
81public:
82 void sort(List<T>& list, std::function<bool(const T& a, const T& b)> comp) override {
83 if (list.size() <= 1) return;
84
85 // Розділяємо на два тимчасових списки нашого типу
86 ArrayList<T> left, right;
87 size_t mid = list.size() / 2;
88 for (size_t i = 0; i < mid; i++) left.add(list.get(i));
89 for (size_t i = mid; i < list.size(); i++) right.add(list.get(i));
90
91 // Рекурсивно сортуємо їх
92 sort(left, comp);
93 sort(right, comp);
94
95 // Зливаємо відсортовані списки назад в оригінальний
96 size_t i = 0, j = 0, k = 0;
97 while (i < left.size() && j < right.size()) {
98 if (comp(left.get(i), right.get(j))) {
99 list.set(k++, left.get(i++));
100 } else {
101 list.set(k++, right.get(j++));
102 }
103 }
104 while (i < left.size()) list.set(k++, left.get(i++));
105 while (j < right.size()) list.set(k++, right.get(j++));
106 }
107};

109template <typename T>
110class HeapSort : public SortingAlgorithm<T> {
111public:
112 void sort(List<T>& list, std::function<bool(const T& a, const T& b)> comparator) override {
113 // Будуємо піраміду (max heap)
114 for (int i = list.size() / 2 - 1; i >= 0; i--) {
115 heapify(list, list.size(), i, comparator);
116 }
117 // Вилучаємо елементи з піраміди
118 for (int i = list.size() - 1; i > 0; i--) {
119 list.swap(0, i);
120 heapify(list, i, 0, comparator);
121 }
122 }
123private:
124 void heapify(List<T>& list, int n, int i, std::function<bool(const T&, const T&)> comp) {
125 int largest = i;
126 int left = 2 * i + 1;
127 int right = 2 * i + 2;
128
129 if (left < n && comp(list.get(largest), list.get(left))) largest = left;
130 if (right < n && comp(list.get(largest), list.get(right))) largest = right;
131
132 if (largest != i) {
133 list.swap(i, largest);
134 heapify(list, n, largest, comp);
135 }
136 }
137};

141class BucketSort : public SortingAlgorithm<T> {
142public:
143 void sort(List<T>& list, std::function<bool(const T&, const T&)> comparator) override {
144 if (list.isEmpty()) return;
145 auto getKey = [](const T& item) { return item->getCharCount(); };
146
147 size_t maxKey = 0;
148 for(size_t i = 0; i < list.size(); ++i) {
149 if (getKey(list.get(i)) > maxKey) maxKey = getKey(list.get(i));
150 }

152 // Використовуємо std::vector для створення кошиків,

154 std::vector<ArrayList<T>> buckets(maxKey + 1);
155
156 for (size_t i = 0; i < list.size(); ++i) {
157 buckets[getKey(list.get(i))].add(list.get(i));
158 }
159
160 size_t current = 0;
161 for (auto& bucket : buckets) {
162 for (size_t i = 0; i < bucket.size(); ++i) {
163 list.set(current++, bucket.get(i));
164 }

170template <typename T>
171class RadixSort : public SortingAlgorithm<T> {
172public:
173 void sort(List<T>& list, std::function<bool(const T&, const T&)> comparator) override {
174 if (list.isEmpty()) return;
175 auto getKey = [](const T& item) { return item->getCharCount(); };
176
177 size_t maxKey = 0;
178 for(size_t i = 0; i < list.size(); ++i) {
179 if (getKey(list.get(i)) > maxKey) maxKey = getKey(list.get(i));
180 }
181
182 for (int exp = 1; maxKey / exp > 0; exp *= 10) {
183 countingSortForRadix(list, exp, getKey);
184 }
185 }
186private:
187 void countingSortForRadix(List<T>& list, int exp, std::function<size_t(const T&)> getKey) {
188 ArrayList<T> output;
189 for(size_t i = 0; i < list.size(); ++i) output.add(list.get(i)); // Потрібен тимчасовий буфер
190
191 int count[10] = {0};
192
193 for (size_t i = 0; i < list.size(); ++i) {
194 count[(getKey(list.get(i)) / exp) % 10]++;
195 }
196 for (int i = 1; i < 10; i++) {
197 count[i] += count[i - 1];
198 }
199 for (int i = list.size() - 1; i >= 0; i--) {
200 output.set(count[(getKey(list.get(i)) / exp) % 10] - 1, list.get(i));
201 count[(getKey(list.get(i)) / exp) % 10]--;
202 }
203 for (size_t i = 0; i < list.size(); i++) {
204 list.set(i, output.get(i));
205 }
206 }
